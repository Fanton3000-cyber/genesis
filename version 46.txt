// === ЭКОНОМИКА: ВСПОМОГАТЕЛЬНАЯ ФУНКЦИЯ BFS ===
function bfsCluster(startStar, factionStars, visited) {
const cluster = [];
const queue = [startStar];
visited.add(startStar.id);
while (queue.length > 0) {
const current = queue.shift();
cluster.push(current);
for (let connId of current.connections) {
const neighbor = factionStars.find(s => s.id === connId);
if (neighbor && !visited.has(neighbor.id)) {
visited.add(neighbor.id);
queue.push(neighbor);
}
}
}
return cluster;
}

// === ЭКОНОМИКА: ПОИСК ЦЕНТРА КЛАСТЕРА ===
function findClusterCenter(cluster) {
if (cluster.length === 0) return null;
let centerStar = cluster[0];
let minAvgDist = Infinity;
for (let star of cluster) {
let totalDist = 0;
for (let other of cluster) {
totalDist += Math.sqrt(Math.pow(star.x - other.x, 2) + Math.pow(star.y - other.y, 2));
}
const avgDist = totalDist / cluster.length;
if (avgDist < minAvgDist) {
minAvgDist = avgDist;
centerStar = star;
}
}
return centerStar;
}

// === ЭКОНОМИКА: ПОИСК КЛАСТЕРОВ ТЕРРИТОРИЙ ===
function findTerritoryClusters(faction) {
const factionStars = state.stars.filter(s => s.factionId === faction.id);
if (factionStars.length === 0) return [];
const clusters = [];
const visited = new Set();
for (let startStar of factionStars) {
if (visited.has(startStar.id)) continue;
const cluster = bfsCluster(startStar, factionStars, visited);
if (cluster.length > 0) {
const centerStar = findClusterCenter(cluster);
const isMain = cluster.some(s => s.id === faction.capitalStarId);
clusters.push({
stars: cluster,
isMain: isMain,
capitalStarId: centerStar.id
});
}
}
return clusters;
}

// === ЭКОНОМИКА: ОБНОВЛЕНИЕ СЧЕТОВ ФРАКЦИИ ===
function updateFactionAccounts(faction) {
const clusters = findTerritoryClusters(faction);
clusters.forEach((cluster, index) => {
let account = faction.accounts.find(a => a.capitalStarId === cluster.capitalStarId);
if (!account) {
account = {
id: faction.accounts.length,
capitalStarId: cluster.capitalStarId,
coins: 0,
stars: cluster.stars.map(s => s.id),
isMain: cluster.isMain
};
faction.accounts.push(account);
if (!cluster.isMain) {
addLog(`${faction.name}: новый изолированный счет (ID: ${account.id}, ${cluster.stars.length} звёзд)`, 'economy');
}
} else {
account.stars = cluster.stars.map(s => s.id);
}
state.stars.forEach(star => {
if (star.factionId === faction.id) {
star.isCapital = (star.id === cluster.capitalStarId && cluster.isMain);
star.isFalseCapital = (star.id === cluster.capitalStarId && !cluster.isMain);
}
});
});
const validCapitalIds = clusters.map(c => c.capitalStarId);
faction.accounts = faction.accounts.filter(acc => {
if (!validCapitalIds.includes(acc.capitalStarId)) {
addLog(`${faction.name}: счет #${acc.id} ликвидирован (${acc.coins} $ потеряно)`, 'warning');
return false;
}
return true;
});
}

// === ЭКОНОМИКА: НАЧИСЛЕНИЕ ДОХОДА ===
function calculateAccountIncome(faction, account) {
const stars = account.stars.map(id => state.stars.find(s => s.id === id)).filter(s => s);
const territoryCount = stars.length;
const starCount = stars.filter(s => !s.isVirtual).length;
const baseIncome = territoryCount * CONFIG.baseIncomePerTerritory;
const starBonus = baseIncome * (starCount * CONFIG.starBonusPercent / 100);
const totalIncome = Math.floor(baseIncome + starBonus);
return {
baseIncome,
starBonus: Math.floor(starBonus),
totalIncome,
territoryCount,
starCount
};
}

function getLineIntersection(p1, p2, p3, p4) {
const det = (p2.x - p1.x) * (p4.y - p3.y) - (p4.x - p3.x) * (p2.y - p1.y);
if (det === 0) return false;
const lambda = ((p4.y - p3.y) * (p4.x - p1.x) + (p3.x - p4.x) * (p4.y - p1.y)) / det;
const gamma = ((p1.y - p2.y) * (p4.x - p1.x) + (p2.x - p1.x) * (p4.y - p1.y)) / det;
return (0 < lambda && lambda < 1) && (0 < gamma && gamma < 1);
}
function countIntersections(newLine, existingLines) {
let count = 0;
for (let line of existingLines) {
if (line.p1.id === newLine.p1.id || line.p1.id === newLine.p2.id ||
line.p2.id === newLine.p1.id || line.p2.id === newLine.p2.id) continue;
if (getLineIntersection({x: newLine.p1.x, y: newLine.p1.y}, {x: newLine.p2.x, y: newLine.p2.y},
{x: line.p1.x, y: line.p1.y}, {x: line.p2.x, y: line.p2.y})) count++;
}
return count;
}

function isPointInPolygon(point, polygon) {
if (polygon.length < 3) return true;
let inside = false;
for (let i = 0, j = polygon.length - 1; i < polygon.length; j = i++) {
const xi = polygon[i].x, yi = polygon[i].y;
const xj = polygon[j].x, yj = polygon[j].y;
const intersect = ((yi > point.y) !== (yj > point.y)) &&
(point.x < (xj - xi) * (point.y - yi) / (yj - yi) + xi);
if (intersect) inside = !inside;
}
return inside;
}
function generateUniverseShape() {

const irregularity = CONFIG.mapShapeIrregularity;

for (let i = 0; i < numPoints; i++) {
const baseAngle = (Math.PI * 2 * i / numPoints);
const radiusVariation = 0.7 + Math.random() * 0.3 * irregularity;
const radius = baseRadius * radiusVariation;
const angleVariation = (Math.random() - 0.5) * irregularity * 0.5;
polygon.push({
x: centerX + Math.cos(baseAngle + angleVariation) * radius,
y: centerY + Math.sin(baseAngle + angleVariation) * radius
});
}
polygon.sort((a, b) => {
const angleA = Math.atan2(a.y - centerY, a.x - centerX);
const angleB = Math.atan2(b.y - centerY, b.x - centerX);
return angleA - angleB;
});
state.mapShapePolygon = polygon;
let minX = Infinity, maxX = -Infinity, minY = Infinity, maxY = -Infinity;
polygon.forEach(p => {
minX = Math.min(minX, p.x); maxX = Math.max(maxX, p.x);

}

function computeVoronoi() {
const bounds = state.universeBounds;
const allStars = [...state.stars, ...state.virtualStars];
allStars.forEach(star => {
star.voronoiCell = [
{x: bounds.minX - 1000, y: bounds.minY - 1000},
{x: bounds.maxX + 1000, y: bounds.minY - 1000},
{x: bounds.maxX + 1000, y: bounds.maxY + 1000},
{x: bounds.minX - 1000, y: bounds.maxY + 1000}
];
});
allStars.forEach(star => {
allStars.forEach(other => {
if (star === other) return;
clipCellByBisector(star, other);
});
});
buildBoundaryFromEdges(allStars);
if (state.boundaryEdges.length > 0) {
let minX = Infinity, maxX = -Infinity, minY = Infinity, maxY = -Infinity;
state.boundaryEdges.forEach(edge => {
minX = Math.min(minX, edge.start.x, edge.end.x);
maxX = Math.max(maxX, edge.start.x, edge.end.x);
minY = Math.min(minY, edge.start.y, edge.end.y);
maxY = Math.max(maxY, edge.start.y, edge.end.y);
});
state.mapBounds = { minX, maxX, minY, maxY, width: maxX - minX, height: maxY - minY };
}
}
function buildBoundaryFromEdges(allStars) {
state.boundaryEdges = [];
const edgeSet = new Set();
state.stars.forEach(realStar => {
if (!realStar.voronoiCell || realStar.voronoiCell.length < 3) return;
for (let i = 0; i < realStar.voronoiCell.length; i++) {
const p1 = realStar.voronoiCell[i];
const p2 = realStar.voronoiCell[(i + 1) % realStar.voronoiCell.length];
const midX = (p1.x + p2.x) / 2;
const midY = (p1.y + p2.y) / 2;
let nearestStar = null, nearestDist = Infinity;
allStars.forEach(other => {
if (other === realStar) return;
const d = Math.sqrt(Math.pow(other.x - midX, 2) + Math.pow(other.y - midY, 2));
if (d < nearestDist) { nearestDist = d; nearestStar = other; }
});
if (nearestStar && nearestStar.isVirtual) {
const key = `${Math.round(p1.x)},${Math.round(p1.y)}-${Math.round(p2.x)},${Math.round(p2.y)}`;
if (!edgeSet.has(key)) {
edgeSet.add(key);
state.boundaryEdges.push({ start: p1, end: p2 });
}
}
}
});
}
function clipCellByBisector(star, other) {
const midX = (star.x + other.x) / 2, midY = (star.y + other.y) / 2;
const nx = star.x - other.x, ny = star.y - other.y;
const newCell = [];
const cell = star.voronoiCell;
if (cell.length === 0) return;
for (let i = 0; i < cell.length; i++) {
const curr = cell[i], next = cell[(i + 1) % cell.length];
const currDist = (curr.x - midX) * nx + (curr.y - midY) * ny;
const nextDist = (next.x - midX) * nx + (next.y - midY) * ny;
if (currDist >= -0.001) newCell.push({x: curr.x, y: curr.y});
if ((currDist >= -0.001 && nextDist < -0.001) || (currDist < -0.001 && nextDist >= -0.001)) {
const t = currDist / (currDist - nextDist);
newCell.push({ x: curr.x + t * (next.x - curr.x), y: curr.y + t * (next.y - curr.y) });
}
}
star.voronoiCell = newCell;
}
function generateConnections() {
state.stars.forEach(star => star.connections = []);
state.routeLines = [];
state.stars.forEach(star => {
const neighbors = [];
state.stars.forEach(other => {
if (star === other) return;
const dx = other.x - star.x, dy = other.y - star.y;
const dist = Math.sqrt(dx * dx + dy * dy);
if (dist < CONFIG.connectionRange && dist > 0) {
neighbors.push({ star: other, dist, angle: Math.atan2(dy, dx) });
}
});
neighbors.sort((a, b) => a.dist - b.dist);
const sectors = new Map();
const sectorSize = Math.PI / 4;
neighbors.forEach(neighbor => {
let sectorIndex = Math.floor((neighbor.angle + Math.PI) / sectorSize);
sectorIndex = Math.max(0, Math.min(7, sectorIndex));
if (!sectors.has(sectorIndex)) sectors.set(sectorIndex, neighbor);
});
Array.from(sectors.values()).sort((a, b) => a.dist - b.dist).slice(0, CONFIG.maxConnections).forEach(neighbor => {
if (star.connections.length >= CONFIG.maxConnections || neighbor.star.connections.length >= CONFIG.maxConnections) return;
const newLine = { p1: star, p2: neighbor.star };
if (countIntersections(newLine, state.routeLines) <= CONFIG.maxRouteIntersections) {
if (!star.connections.includes(neighbor.star.id)) {
star.connections.push(neighbor.star.id);
neighbor.star.connections.push(star.id);
state.routeLines.push(newLine);
}
}
});
});
ensureConnectivity();
}
function ensureConnectivity() {
const connected = new Set([0]);
while (connected.size < state.stars.length) {
let minDist = Infinity, bestPair = null;
for (let i of connected) {
for (let j = 0; j < state.stars.length; j++) {
if (connected.has(j)) continue;
if (state.stars[i].connections.length >= CONFIG.maxConnections || state.stars[j].connections.length >= CONFIG.maxConnections) continue;
const d = Math.sqrt(Math.pow(state.stars[i].x - state.stars[j].x, 2) + Math.pow(state.stars[i].y - state.stars[j].y, 2));
if (d < minDist) { minDist = d; bestPair = [i, j]; }
}
}
if (bestPair) {
const [i, j] = bestPair;
if (!state.stars[i].connections.includes(j)) {
state.stars[i].connections.push(j);
state.stars[j].connections.push(i);
}
connected.add(j);
} else break;
}
}
function generateStarsInUniverse() {
const polygon = state.mapShapePolygon;
const bounds = state.universeBounds;
if (polygon.length < 3) return;
const insideCount = Math.floor(state.starCount * 0.8);
let attempts = 0, maxAttempts = state.starCount * 500;
while (state.stars.length < insideCount && attempts < maxAttempts) {
const x = bounds.minX + Math.random() * (bounds.maxX - bounds.minX);
const y = bounds.minY + Math.random() * (bounds.maxY - bounds.minY);
if (!isPointInPolygon({x, y}, polygon)) { attempts++; continue; }
let tooClose = false;
for (let star of state.stars) {
if (Math.sqrt(Math.pow(star.x - x, 2) + Math.pow(star.y - y, 2)) < CONFIG.minStarDistance) {
tooClose = true; break;
}
}
if (!tooClose) state.stars.push(new Star(state.stars.length, x, y, false));
attempts++;
}
}

case 1: x = bounds.maxX + offset; y = bounds.minY + Math.random() * mapHeight; break;
case 2: x = bounds.minX + Math.random() * mapWidth; y = bounds.maxY + offset; break;
case 3: x = bounds.minX - offset; y = bounds.minY + Math.random() * mapHeight; break;
}
state.virtualStars.push(new Star(state.stars.length + i, x, y, true));
}
}

function updatePlayer(deltaTime) {

player.moveProgress += (CONFIG.playerMoveSpeed * deltaTime) / player.moveDistance;
if (player.moveProgress >= 1) {
player.currentStarId = player.targetStarId;
player.targetStarId = null;
player.originStarId = null;
player.isMoving = false;
player.moveProgress = 0;
player.moveDistance = 0;
gameTimer.isRunning = false;
} else {
player.position.x = player.startPos.x + (player.targetOrbitPoint.x - player.startPos.x) * player.moveProgress;
player.position.y = player.startPos.y + (player.targetOrbitPoint.y - player.startPos.y) * player.moveProgress;
}
}
} else {
player.orbitAngle += CONFIG.playerOrbitSpeed * deltaTime;
player.position.x = currentStar.x + Math.cos(player.orbitAngle) * CONFIG.playerOrbitRadius;
player.position.y = currentStar.y + Math.sin(player.orbitAngle) * CONFIG.playerOrbitRadius;
}
}

}
}
turnTimerBar.style.width = `${(gameTimer.current / gameTimer.max) * 100}%`;
}
function drawCapitalPulses() {
const currentTime = state.time * 1000;
if (currentTime - state.lastPulseTime >= CONFIG.capitalPulseInterval) {
state.stars.forEach(star => {
if ((star.isCapital || star.isFalseCapital) && star.factionId !== null) {
capitalPulses.push({
starId: star.id,
factionId: star.factionId,
birthTime: currentTime,
age: 0,
isFalseCapital: star.isFalseCapital
});
}
});
state.lastPulseTime = currentTime;
}
for (let i = capitalPulses.length - 1; i >= 0; i--) {
const pulse = capitalPulses[i];
pulse.age = currentTime - pulse.birthTime;
if (pulse.age > CONFIG.capitalPulseDuration) {
capitalPulses.splice(i, 1);
continue;
}
const star = state.stars.find(s => s.id === pulse.starId);
if (!star || !factions.list[pulse.factionId]) continue;
const progress = pulse.age / CONFIG.capitalPulseDuration;
const alpha = 0.5 * (1 - progress);
const scale = 1 + progress * 1.5;
ctx.save();
ctx.translate(star.x, star.y);
ctx.globalAlpha = alpha;
ctx.fillStyle = factions.list[pulse.factionId].color.border;
const baseSize = (14 / camera.zoom) * scale;
const innerRatio = 0.4;
ctx.beginPath();
for (let j = 0; j < 10; j++) {
const angle = (Math.PI * 2 * j / 10) - Math.PI / 2;
const radius = (j % 2 === 0) ? baseSize : baseSize * innerRatio;
const x = Math.cos(angle) * radius;
const y = Math.sin(angle) * radius;
if (j === 0) ctx.moveTo(x, y);
else ctx.lineTo(x, y);
}
ctx.closePath();
if (pulse.isFalseCapital) {
ctx.strokeStyle = factions.list[pulse.factionId].color.border;
ctx.lineWidth = 3 / camera.zoom;
ctx.stroke();
ctx.beginPath();
ctx.arc(0, 0, baseSize * 0.5, 0, Math.PI * 2);
ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
ctx.fill();
} else {
ctx.fill();
}
ctx.restore();
}
}
function drawFactionEmblems() {
factions.list.forEach(faction => {
const clusters = findTerritoryClusters(faction);
clusters.forEach(cluster => {
const capitalStar = state.stars.find(s => s.id === cluster.capitalStarId);
if (!capitalStar) return;
let minX = Infinity, maxX = -Infinity, minY = Infinity, maxY = -Infinity;
cluster.stars.forEach(star => {
if (star.voronoiCell && star.voronoiCell.length > 0) {
star.voronoiCell.forEach(point => {
minX = Math.min(minX, point.x); maxX = Math.max(maxX, point.x);
minY = Math.min(minY, point.y); maxY = Math.max(maxY, point.y);
});
}
});
if (minX === Infinity) return;
const centerX = (minX + maxX) / 2, centerY = (minY + maxY) / 2;
const emblemSize = Math.min(maxX - minX, maxY - minY) * 0.35;
ctx.save();
ctx.translate(centerX, centerY);
ctx.globalAlpha = 0.35;
const svgPath = faction.emblem.match(/d="([^"]+)"/);
if (svgPath && svgPath[1]) {
const path2D = new Path2D(svgPath[1]);
ctx.fillStyle = faction.color.border;
ctx.scale(emblemSize / 24, emblemSize / 24);
ctx.translate(-12, -12);
ctx.fill(path2D);
}
ctx.restore();
});
});
}
function drawMap() {
ctx.clearRect(0, 0, canvas.width, canvas.height);
ctx.save();
ctx.translate(canvas.width / 2, canvas.height / 2);
ctx.scale(camera.zoom, camera.zoom);
ctx.translate(-canvas.width / 2, -canvas.height / 2);
ctx.translate(-camera.x, -camera.y);
state.stars.forEach(star => {
if (!star.voronoiCell || star.voronoiCell.length < 3) return;
if (star.factionId !== null && factions.list[star.factionId]) {
ctx.fillStyle = factions.list[star.factionId].color.main;
ctx.strokeStyle = factions.list[star.factionId].color.border;
} else {
ctx.fillStyle = 'rgba(50, 50, 50, 0.1)';
ctx.strokeStyle = CONFIG.colors.sectorBorder;
}
ctx.lineWidth = 1 / camera.zoom;
ctx.beginPath();
ctx.moveTo(star.voronoiCell[0].x, star.voronoiCell[0].y);
for (let i = 1; i < star.voronoiCell.length; i++)
ctx.lineTo(star.voronoiCell[i].x, star.voronoiCell[i].y);
ctx.closePath();
ctx.fill();
ctx.stroke();
});
ctx.lineWidth = 1.5 / camera.zoom;
state.stars.forEach(star => {
star.connections.forEach(connId => {
if (connId < star.id) return;
const target = state.stars.find(s => s.id === connId);
if (target) {
const brightness = (star.brightness + target.brightness) / 2;
ctx.strokeStyle = brightness > 0.85 ? CONFIG.colors.lineBright : CONFIG.colors.line;
ctx.globalAlpha = 0.5 + brightness * 0.5;
ctx.beginPath();
ctx.moveTo(star.x, star.y);
ctx.lineTo(target.x, target.y);
ctx.stroke();
ctx.globalAlpha = 1.0;
}
});
});
if (!player.isMoving && player.currentStarId !== null) {
const currentStar = state.stars.find(s => s.id === player.currentStarId);
if (currentStar) {
currentStar.connections.forEach(connId => {
const target = state.stars.find(s => s.id === connId);
if (target) {
ctx.strokeStyle = CONFIG.colors.lineAvailable;
ctx.globalAlpha = 0.8;
ctx.lineWidth = 2.5 / camera.zoom;
ctx.beginPath();
ctx.moveTo(currentStar.x, currentStar.y);
ctx.lineTo(target.x, target.y);
ctx.stroke();
ctx.globalAlpha = 1.0;
ctx.lineWidth = 1.5 / camera.zoom;
}
});
}
}
state.stars.forEach(star => {
star.angle += 0.015;
ctx.save();
ctx.translate(star.x, star.y);
if ((star.isCapital || star.isFalseCapital) && factions.list[star.factionId]) {
ctx.globalAlpha = 0.9;
ctx.fillStyle = factions.list[star.factionId].color.border;
const capitalSize = 14 / camera.zoom;
const innerRatio = 0.4;
ctx.beginPath();
for (let i = 0; i < 10; i++) {
const angle = (Math.PI * 2 * i / 10) - Math.PI / 2;
const radius = (i % 2 === 0) ? capitalSize : capitalSize * innerRatio;
const x = Math.cos(angle) * radius;
const y = Math.sin(angle) * radius;
if (i === 0) ctx.moveTo(x, y);
else ctx.lineTo(x, y);
}
ctx.closePath();
if (star.isFalseCapital) {
ctx.strokeStyle = factions.list[star.factionId].color.border;
ctx.lineWidth = 3 / camera.zoom;
ctx.stroke();
ctx.beginPath();
ctx.arc(0, 0, capitalSize * 0.5, 0, Math.PI * 2);
ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
ctx.fill();
} else {
ctx.fill();
}
ctx.globalAlpha = 1.0;
}
ctx.globalAlpha = 0.25;
ctx.save();
ctx.rotate(star.angle);
if (star.starType === 'red') ctx.fillStyle = CONFIG.colors.starRed;
else if (star.starType === 'blue') ctx.fillStyle = CONFIG.colors.starBlue;
else ctx.fillStyle = CONFIG.colors.starYellow;
const size = 5 / camera.zoom;
ctx.fillRect(-size, -size, size * 2, size * 2);
ctx.restore();
ctx.globalAlpha = 1.0;
ctx.fillStyle = star.isPlayer ? '#fff' : (star.starType === 'red' ? CONFIG.colors.starRed : (star.starType === 'blue' ? CONFIG.colors.starBlue : CONFIG.colors.starYellow));
ctx.fillRect(-3/camera.zoom, -3/camera.zoom, 6/camera.zoom, 6/camera.zoom);
ctx.restore();
});
drawCapitalPulses();
drawFactionEmblems();
ctx.save();
ctx.translate(player.position.x, player.position.y);
let arrowAngle = player.isMoving && player.targetStarId ?
Math.atan2(state.stars.find(s => s.id === player.targetStarId).y - player.position.y,
state.stars.find(s => s.id === player.targetStarId).x - player.position.x) :
player.orbitAngle + Math.PI / 2;
ctx.rotate(arrowAngle + Math.PI / 2);
const arrowSize = 12 / camera.zoom;
ctx.fillStyle = CONFIG.colors.player;
ctx.shadowBlur = 10 / camera.zoom;
ctx.shadowColor = CONFIG.colors.player;
ctx.beginPath();
ctx.moveTo(0, -arrowSize);
ctx.lineTo(arrowSize * 0.6, arrowSize * 0.5);
ctx.lineTo(0, arrowSize * 0.2);
ctx.lineTo(-arrowSize * 0.6, arrowSize * 0.5);
ctx.closePath();
ctx.fill();
ctx.shadowBlur = 0;
ctx.restore();
ctx.restore();
drawDeepSpace();
}
let lastTime = 0;
function gameLoop(currentTime) {
if (state.isRunning) {
const deltaTime = currentTime - lastTime;
lastTime = currentTime;
state.time += 0.05;
updatePlayer(deltaTime);
updateTimer(deltaTime);
drawMap();
}
requestAnimationFrame(gameLoop);
}
function startGame() {
if (state.isRunning || state.isGenerating) return;
state.isGenerating = true;
loadingIndicator.style.display = 'block';
startButton.disabled = true;
startButton.textContent = 'ГЕНЕРАЦИЯ...';
starSlider.disabled = true;
setTimeout(() => {
state.isRunning = true;
state.isGenerating = false;
startScreen.style.display = 'none';
statusLight.classList.add('active');
sysStatusText.innerText = "ONLINE";
sysStatusText.style.color = "#00e676";
resizeCanvas();
generateMap();
generateBackgroundStars();
gameTimer.current = CONFIG.turnTimeLimit;
gameTimer.turnCount = 0;
gameTimer.isRunning = false;
turnCounterValue.textContent = '0';
turnTimerBar.style.width = '100%';
lastTime = performance.now();
requestAnimationFrame(gameLoop);
}, 100);
}
window.addEventListener('keydown', (e) => {
if (e.code === 'Space') {
e.preventDefault();
if (!state.isRunning && !state.isGenerating) startGame();
}
});
startButton.addEventListener('click', (e) => {
e.stopPropagation();
if (!state.isRunning && !state.isGenerating) startGame();
});
resizeCanvas();
